# JSON.parse()
- JSON.parse(string-object) method is used to convert json string to js object.

ex: 
const str='{'a':2}'
const obj={a:2};

# Object.keys()
- Object.keys (obj) method returns a array containing the keys of objects.

ex:
const obj={a:2,b:5};
console.log(Object.keys(obj));

output: [ 'a', 'b' ]
